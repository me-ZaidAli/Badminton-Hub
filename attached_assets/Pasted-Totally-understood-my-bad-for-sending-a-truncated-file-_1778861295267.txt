    function calculateTeamAverageDiff(teamA: Player[], teamB: Player[]): number {

    function calculateFairnessScore(group: Player[], history: PairingHistory, settings: MatchmakingSettings): number {
        let score = 0;

        // Deficit penalty: prioritise players with higher games behind
        const maxDeficit = Math.max(...group.map(p => p.deficit || 0));
        score += maxDeficit * (settings.deficitWeight || 2);

        // Games played penalty: avoid overusing same players
        const totalGames = group.reduce((sum, p) => sum + (p.gamesPlayed || 0), 0);
        score += totalGames * 0.5;

        // Partner repeat penalty
        const repeatPartners = countRepeatPartners(group, history);
        score += repeatPartners * (settings.partnerRepeatWeight || 5);

        // Group repeat penalty
        const repeatGroups = countRepeatGroups(group, history);
        score += repeatGroups * (settings.groupRepeatPenalty || 8);

        // Grade spread penalty
        const spread = calculateGradeSpread(group);
        if (spread > settings.gradeSpreadLimit) {
            score += (spread - settings.gradeSpreadLimit) * 10;
        }

        return score;
    }

    function countRepeatPartners(group: Player[], history: PairingHistory): number {

    function classifyMatchType(teamA: Player[], teamB: Player[]): 'MALE_ONLY' | 'FEMALE_ONLY' | 'MIXED' {
        const allPlayers = [...teamA, ...teamB];
        const hasMale = allPlayers.some(p => p.gender === 'M');
        const hasFemale = allPlayers.some(p => p.gender === 'F');

    export function buildPairingHistory(previousMatches: MatchResult[] = []): PairingHistory {
        const playerHistory = new Map<number, PlayerMatchHistory>();
        const recentGroups = new Set<Set<number>>();

        previousMatches.forEach(match => {
            // Build per-player history
            match.players.forEach(p => {
                if (!playerHistory.has(p.id)) playerHistory.set(p.id, { matches: [] });
                const entry = playerHistory.get(p.id)!;
                entry.matches.push({
                    timestamp: match.timestamp,
                    type: match.type,
                    partnerIds: match.players.filter(op => op.id !== p.id).map(op => op.id)
                });
            });

            // Track full groups for repeat prevention
            recentGroups.add(new Set(match.players.map(p => p.id)));
        });

        return { playerHistory, recentGroups };
    }

    export function generateSmartMatches(
        players: Player[],
        previousHistory: PairingHistory,
        settings: MatchmakingSettings,
        mode: 'DOUBLES' | 'SINGLES' = 'DOUBLES',
        matchCount: number = 3,
        fixedPairs?: Array<[number, number]>
    ): MatchResult[] {
        // Guard against empty input
        if (!players.length) return [];

        // Normalise settings against presets
        const activeSettings = {...PRESETS[settings.preset || 'balanced'], ...settings};

        return mode === 'DOUBLES'
            ? generateDoublesMatches(players, previousHistory, activeSettings, matchCount, fixedPairs)
            : generateSinglesMatches(players, previousHistory, activeSettings, matchCount);
    }

    export async function tryGenerate(
        players: Player[],
        history: PairingHistory,
        settings: MatchmakingSettings,
        mode: 'DOUBLES' | 'SINGLES',
        attempts: number = 3
    ): Promise<MatchResult[]> {
        let best: MatchResult[] = [];
        let bestScore = Infinity;

        for (let i = 0; i < attempts; i++) {
            const result = generateSmartMatches(players, history, settings, mode);
            if (!result.length) continue;

            const totalScore = result.reduce((sum, m) => sum + (m.fairnessScore || 999), 0);
            if (totalScore < bestScore) {
                best = result;
                bestScore = totalScore;
            }
        }

        return best;
    }

    // Helper: Generate combinations with early exit
    function selectUnits<T>(arr: T[], size: number, limit: number): T[][] {
        const results: T[][] = [];
        const max = Math.min(limit, 5000);

        function combine(start: number, current: T[]) {
            if (results.length >= max) return;
            if (current.length === size) {
                results.push([...current]);
                return;
            }
            for (let i = start; i < arr.length; i++) {
                current.push(arr[i]);
                combine(i + 1, current);
                current.pop();
            }
        }

        combine(0, []);
        return results;
    }
